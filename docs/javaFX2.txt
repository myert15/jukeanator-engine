✅ 1. JavaFX Main App (bootstraps Spring)
package com.djt.jukeanator_engine.ui;

import javafx.application.Application;
import javafx.application.Platform;
import javafx.fxml.FXMLLoader;
import javafx.scene.Scene;
import javafx.stage.Stage;

import org.springframework.boot.builder.SpringApplicationBuilder;
import org.springframework.context.ConfigurableApplicationContext;

import com.djt.jukeanator_engine.JukeanatorEngineApplication;

public class JavaFxApp extends Application {

    private ConfigurableApplicationContext context;

    @Override
    public void init() {
        context = new SpringApplicationBuilder(JukeanatorEngineApplication.class)
                .run();
    }

    @Override
    public void start(Stage stage) throws Exception {
        FXMLLoader loader = new FXMLLoader(getClass().getResource("/fxml/main-view.fxml"));
        loader.setControllerFactory(context::getBean);

        Scene scene = new Scene(loader.load(), 900, 600);

        stage.setTitle("JukeANator");
        stage.setScene(scene);
        stage.show();
    }

    @Override
    public void stop() {
        context.close();
        Platform.exit();
    }

    public static void main(String[] args) {
        launch(args);
    }
}

✅ 3. Main JavaFX Controller
package com.djt.jukeanator_engine.ui;

import javafx.application.Platform;
import javafx.collections.FXCollections;
import javafx.fxml.FXML;
import javafx.scene.control.*;

import org.springframework.stereotype.Component;

import com.djt.jukeanator_engine.client.SongLibraryClient;
import com.djt.jukeanator_engine.domain.songlibrary.dto.*;

import java.util.List;

@Component
public class MainViewController {

    private final SongLibraryClient client;

    public MainViewController(SongLibraryClient client) {
        this.client = client;
    }

    @FXML private ListView<GenreDto> genreList;
    @FXML private ListView<ArtistDto> artistList;
    @FXML private ListView<AlbumDto> albumList;
    @FXML private Button refreshButton;

    @FXML
    public void initialize() {
        setupCellFactories();
        loadGenres();
    }

    private void setupCellFactories() {
        genreList.setCellFactory(lv -> new ListCell<>() {
            @Override protected void updateItem(GenreDto item, boolean empty) {
                super.updateItem(item, empty);
                setText(empty ? null : item.getName());
            }
        });

        artistList.setCellFactory(lv -> new ListCell<>() {
            @Override protected void updateItem(ArtistDto item, boolean empty) {
                super.updateItem(item, empty);
                setText(empty ? null : item.getName());
            }
        });

        albumList.setCellFactory(lv -> new ListCell<>() {
            @Override protected void updateItem(AlbumDto item, boolean empty) {
                super.updateItem(item, empty);
                setText(empty ? null : item.getName());
            }
        });
    }

    @FXML
    private void loadGenres() {
        runAsync(() -> {
            List<GenreDto> genres = client.getGenres();
            Platform.runLater(() ->
                genreList.setItems(FXCollections.observableArrayList(genres))
            );
        });
    }

    @FXML
    private void onRefresh() {
        loadGenres();
    }

    @FXML
    private void onGenreSelected() {
        GenreDto selected = genreList.getSelectionModel().getSelectedItem();
        if (selected == null) return;

        runAsync(() -> {
            List<ArtistDto> artists = client.getArtists();
            Platform.runLater(() ->
                artistList.setItems(FXCollections.observableArrayList(artists))
            );
        });
    }

    @FXML
    private void onArtistSelected() {
        ArtistDto selected = artistList.getSelectionModel().getSelectedItem();
        if (selected == null) return;

        runAsync(() -> {
            List<AlbumDto> albums = client.getAlbums();
            Platform.runLater(() ->
                albumList.setItems(FXCollections.observableArrayList(albums))
            );
        });
    }

    private void runAsync(Runnable task) {
        new Thread(() -> {
            try {
                task.run();
            } catch (Exception e) {
                e.printStackTrace();
                Platform.runLater(() -> showError(e));
            }
        }).start();
    }

    private void showError(Exception e) {
        Alert alert = new Alert(Alert.AlertType.ERROR);
        alert.setHeaderText("Error");
        alert.setContentText(e.getMessage());
        alert.showAndWait();
    }
}

<?xml version="1.0" encoding="UTF-8"?>

<HBox xmlns:fx="http://javafx.com/fxml"
      fx:controller="com.djt.jukeanator_engine.ui.MainViewController"
      spacing="10" prefWidth="900" prefHeight="600">

    <VBox spacing="5">
        <Label text="Genres"/>
        <ListView fx:id="genreList" onMouseClicked="#onGenreSelected"/>
    </VBox>

    <VBox spacing="5">
        <Label text="Artists"/>
        <ListView fx:id="artistList" onMouseClicked="#onArtistSelected"/>
    </VBox>

    <VBox spacing="5">
        <Label text="Albums"/>
        <ListView fx:id="albumList"/>
    </VBox>

    <VBox spacing="10">
        <Button fx:id="refreshButton" text="Refresh" onAction="#onRefresh"/>
    </VBox>

</HBox>
